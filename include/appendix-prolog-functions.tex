\appendix\section{Useful prolog functions}
\subsection{Listen}
\begin{minted}[breaklines]{prolog}
member(X,L) /** is X in L **/
append(S1,S2,S) /** S1 ++ S2 == S **/
rev(X,Y) /** reverse **/
permute(P,X) /** permutations of P **/
\end{minted}
